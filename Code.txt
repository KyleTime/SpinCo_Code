//Follow Right
player.onChat("run", function () {
    while(!motocart.isSidewalkRight())
    {
        motocart.rotate(90)
    }
    
    while(true)
    {
        motocart.move(1)
        if(!motocart.isSidewalkRight())
        {
            motocart.rotate(90)
        }
        else if(motocart.isSidewalkAhead())
        {
            motocart.rotate(-90)
        }
        if(motocart.isAtStop())
        {
            motocart.pauseAtStopSign()
        }
    }
})
